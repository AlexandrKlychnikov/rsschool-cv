//accum("RqaEzty") -> "R-Qq-Aaa-Eeee-Zzzzz-Tttttt-Yyyyyyy"          
function accum(str) {
  let newS = str.split('').reduce((acc, cur, i) => {
    let letter = ''
    for (let n=1; n <= i+1; n++) {
      letter += cur; 
    } 
    let newLetter = letter[0].toUpperCase() + letter.toLowerCase().slice(1)
    acc.push(newLetter);
    return acc;
  }, [])
  return newS.join('-');
}
